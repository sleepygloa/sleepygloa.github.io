## JS - 랜덤
```javascript
window.onload = function(){
  
  var button = document.querySelector(".startBtn");
  var h1 = document.getElementsByTagName("h1")[0];

  var handArr = ["가위", "바위", "보"];
  
  button.addEventListener("click", function(){
    var num = Math.round(Math.random() * (handArr.length-1)); 
    // var num = Math.round( Math.random() * 2);
    h1.innerHTML = handArr[num];
    // alert(handArr[num])
  })
}
```

```html
<button type="button" class="startBtn">리셋</button>
<h1>가위</h1>
```

> 표현할 문구를 배열에 넣고(***handArr***)
> ***Math.round(Math.random() * (handArr.length-1));***
> 을 이용하여 배열의 크기만큼의 랜덤 수를 뽑아 무작위의 문구를 표현.

```javascript
window.onload = function(){
  var resetBtn = document.querySelector(".resetBtn");
  var startBtn = document.querySelector(".startBtn");

  var h1 = document.getElementsByTagName("h1")[0];
  var arrow = document.querySelector(".arrow");
  var rotateNum = 0;

  //초기화
  resetBtn.addEventListener("click", function(){
    rotateNum = 0;
    h1.innerHTML = rotateNum + "deg";
    arrow.style.transform = "rotate("+ rotateNum +"deg)";
  })

  startBtn.addEventListener("click", function(){
    rotateNum = Math.round(Math.random()*720);
    // alert(rotateNum)
    h1.innerHTML = rotateNum + "deg";

    arrow.style.transform = "rotate("+ rotateNum +"deg)";

  })
}
```

```html
<button type="button" class="resetBtn">리셋</button>
<button type="button" class="startBtn">회전</button>
<h1>0deg</h1>
<div class="arrowWrap">
  <img src="./image/up-arrow.svg" alt="arrow" class="arrow">
</div>
```

1. 리셋을 하면 화살표가 초기 위치로 돌아가고
2. 회전을 누르면 랜덤한 위치로 화살표가 회전한다.
+ ```transition : transform 2s cubic-bezier(1.000, -0.010, 0.175, 0.970);``` 옵션으로 화살표가 스무스 하게 이동한다.
+ ```transition: all .2s ease-in-out;``` 옵션은 획획 하고 이동한다.

```javascript

window.onload = function(){
  var input = document.getElementsByTagName("input")[0];
  var result = document.querySelector(".result");
  
  var button = document.querySelector(".startBtn");
  button.addEventListener("click", function(){
    resultFunc();
  });

  var resultArr = ["업 ▲", "다운 ▼", "정답!"]
  var randomNum = returnNum(50);//Math.round(Math.random()*50);

  function resultFunc(){
    var value = input.value;

    if(value < randomNum){
      result.innerHTML = resultArr[0];

    }else if(value > randomNum){
      result.innerHTML = resultArr[1];

    }else{
      //정답
      result.innerHTML = resultArr[2];
      alert("정답입니다.")
      //2초뒤 리셋
      setTimeout(function(){
        resetFunc();
      }, 2000)
    }
    
  }

  //초기화
  function resetFunc(){
    input.value = 1;
    result.innerHTML = "한 판더! (1부터 50사이 정수)";
    // randomNum = Math.round(Math.random()*50);
    randomNum = returnNum(50);
  }

  //랜덤 넘버 리턴
  function returnNum(num){
    return Math.round(Math.random()*num);
  }

}
```

```html
<div class="contWrap">
  <h2>내가 생각한 숫자를 맞춰보세요!</h2>
  <p class="result">(힌트 : 1부터 50사이 정수)</p>
  <input type="number" value="1" max="50" min="1">
  <button type="button" class="startBtn">확인</button>
</div>
```

1. 프로그램 로딩시 (document.onload) 랜덤한 숫자가 정해진다.
2. 화면의 숫자를 입력하여 확인 버튼을 누르면 숫자를 비교하여 정담인지, 높은지, 낮은지 알려준다.

```javascript
window.onload = function(){
  var input = document.getElementsByTagName("input")[0];
  var sec = document.querySelector(".sec");
  
  var button = document.querySelector(".startBtn");
  button.addEventListener("click", function(){
    interval();
  });

  var timer;

  function interval(){
    clearInterval(timer);

    value = input.value * 10000;
    secValue = value / 365 / 24 / 60 / 60;
    
    var secNum = 0;
    timer = setInterval(function(){
      secNum ++;
      // console.log(secNum, secValue * secNum)
      // sec.innerHTML = (secValue * secNum).toFixed(1) +"원";
      sec.innerHTML = "당신은 "+secNum+"초에 "+ (secValue * secNum).toFixed(1) +"원 벌고 있습니다";
    }, 1000)
  }
}
```

```html

  <h3>연봉을 입력해보세요(만)</h3>
  <input type="number" value="5000">
  <button type="button" class="startBtn">확인</button>

  <p class="sec">당신은 1초에 얼마를 벌고 있을까요?</p>
</div>
```

1. 자신의 연봉을 기입하여 확인을 누르면 초당 버는 금액이 1초마다 표시된다.

+ 시작시 clearInterval 을 꼭 해주어야한다.

```javascript
window.onload = function(){
  var prevBtn = document.querySelector(".prevBtn");
  var nextBtn = document.querySelector(".nextBtn");
  var h1 = document.getElementsByTagName("h1")[0];
  var bgColorArr = ["#85FFBD", "#FFFB7D", "#E0C3FC", "#00DBDE"];

  var totalNum = bgColorArr.length;
  var pageNum = Math.round(Math.random()*(totalNum-1));

  prevBtn.addEventListener("click", function(){
    if(pageNum > 0){
      pageNum --;
    }else{
      pageNum = totalNum-1;
    }
    bgColorChange();
  })
  nextBtn.addEventListener("click", function(){
    if(pageNum < totalNum-1){
      pageNum ++;
    }else {
      pageNum = 0;
    }
    bgColorChange();
  })

  function bgColorChange(){
    h1.innerHTML = pageNum + " 페이지 입니다.";
    document.getElementsByTagName("body")[0].style.background = bgColorArr[pageNum];
  }

  //처음 실행
  bgColorChange();
}
```


```html
<button type="button" class="prevBtn">PREV</button>
<button type="button" class="nextBtn">NEXT</button>
```

- NEXT, PREV 버튼을 누르면 배경이 배열의 RGB 코드 대로 변경된다.
- CSS ```body{ transition : background 2.5s ease-out; }``` 을 이용하여 배경이 변경된다.

window.onload = function(){
  var prevBtn = document.querySelector(".prevBtn");
  var nextBtn = document.querySelector(".nextBtn");
  var h1 = document.getElementsByTagName("h1")[0];
  var bgColorArr = ["#85FFBD", "#FFFB7D", "#E0C3FC", "#00DBDE"];
  var section = document.getElementsByTagName("section");

  var pageNum = 0;
  var totalNum = section.length;

  prevBtn.addEventListener("click", function(){
    if(pageNum > 0){
      pageNum --;
    }
    pageChangeFunc();
    window.scrollTo({
      top: section[pageNum].offsetTop,
      behavior: 'smooth',
    })
  })

      pageNum ++;
    }
    pageChangeFunc();

      top: section[pageNum].offsetTop,
      behavior: 'smooth',
    })
  })

  function pageChangeFunc(){
    h1.innerHTML = (pageNum + 1) + " 페이지 입니다.";

    // console.log(bgColorArr[pageNum])
  }

  //페이지 로드되면 바로 실행
  pageChangeFunc();


  window.addEventListener("scroll", function(event){
    var scroll = this.scrollY;
    // var scroll2 = document.documentElement.scrollTop;
    // console.log(scroll, scroll2)

    //첫번째 방법. 
    // 1차원 적이지만 페이지 수가 2개 정도면 괜찮다.
    // if(scroll < section[1].offsetTop){

    // }
    // if(scroll > section[1].offsetTop){
    //   pageNum = 1;
    // }
    // if(scroll > section[2].offsetTop){
    //   pageNum = 2;
    // }
    // if(scroll > section[3].offsetTop){
    //   pageNum = 3;
    // }


    //두번째 방법
    for(var i=0; i<totalNum; i++){

  });
}
